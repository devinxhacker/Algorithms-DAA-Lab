#include<iostream>
#include<vector>
#include<cstdlib>
#include<fstream>
#include<chrono>
using namespace std;
using namespace std::chrono;

class Sort {
    private:
        int n;
        vector<int> arr;
    public:
        Sort(int size) {
            n = size;
            arr.resize(n);
        }
        void upload(string filename);
        void generateRandom(string filename);
        void quickSortRand(int left, int right);
        void quickSortReg(int left, int right);
        void display();
        
};

void Sort::upload(string filename) {
    freopen(filename.c_str(), "r", stdin);
    for(int i = 0; i < n; i++) {
        cin >> arr[i];
    }
}
void Sort::generateRandom(string filename) {
    ofstream file(filename);
    for(int i = 0; i < n; i++) {
        arr[i] = rand() % 1000;
        file << arr[i] << " ";
    }
    file.close();
}
void Sort::quickSortRand(int left, int right){
    if(left < right) {
        // randomly select a pivot
        int pivotIndex = left + rand() % (right - left + 1);
        int pivotValue = arr[pivotIndex];
        
        // partition the array around the pivot
        int i = left, j = right;
        while(i <= j) {
            while(arr[i] < pivotValue) i++;
            while(arr[j] > pivotValue) j--;
            if(i <= j) {
                swap(arr[i], arr[j]);
                i++;
                j--;
            }
        }
        
        // recursively sort the partitions
        quickSortRand(left, j);
        quickSortRand(i, right);
    }
}

void Sort::quickSortReg(int left, int right){
    if(left < right) {
        // select the first element as pivot
        int pivotIndex = left;
        int pivotValue = arr[pivotIndex];
        
        // partition the array around the pivot
        int i = left, j = right;
        while(i <= j) {
            while(arr[i] < pivotValue) i++;
            while(arr[j] > pivotValue) j--;
            if(i <= j) {
                swap(arr[i], arr[j]);
                i++;
                j--;
            }
        }
        
        // recursively sort the partitions
        quickSortReg(left, j);
        quickSortReg(i, right);
    }
}
void Sort::display() {
    for(int i = 0; i < n; i++) {
        cout << arr[i] << " ";
    }
    cout << endl;
}

int main() {
    int n;
    cout << "Enter number of elements: ";
    cin >> n;
    
    Sort sorter(n);
    
    string filename = "testcase.txt";
    sorter.generateRandom(filename);
    sorter.upload(filename);
    

    
    // random pivot selection
    auto start = high_resolution_clock::now();
    sorter.quickSortRand(0, n - 1);
    auto end = high_resolution_clock::now();
    
    // Output
    cout << "Sorted array (Random Pivot): ";
    sorter.display();
    auto duration = duration_cast<nanoseconds>(end - start);
    cout << "Execution time (Random Pivot): " << duration.count() << " nanoseconds" << endl;



    // regular pivot selection
    start = high_resolution_clock::now();
    sorter.quickSortReg(0, n - 1);
    end = high_resolution_clock::now();
    
    // Output
    // cout << "Sorted array (Regular Pivot): ";
//     sorter.display();
    duration = duration_cast<nanoseconds>(end - start);
    cout << "Execution time (Regular Pivot): " << duration.count() << " nanoseconds" << endl;
    
    return 0;
}
